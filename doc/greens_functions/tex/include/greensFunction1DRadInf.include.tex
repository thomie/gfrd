\subsection{1D Rad Inf solution}

\paragraph{Green's function}
\begin{multline}
  v(z,t|z',t'=0) = \frac{1}{\sqrt{4\pi D_1 t}} [
  e^{-(z-z')^2/(4 D_1 t)} +
  e^{-(z+z')^2/(4 D_1 t)} ] \\
  -h e^{h(z+z') + D_1 th^2} \erfc(\frac{z+z'}{2\sqrt{D_1t}} + h\sqrt{D_1t}).
\end{multline}

  \erfc(\frac{z'}{2 \sqrt{D_1 t}}).
