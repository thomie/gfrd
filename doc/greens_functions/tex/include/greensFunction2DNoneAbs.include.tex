\subsection{2D None Abs solution}

\paragraph{Green's function}
\begin{align}
  v(r,\t,t) = \frac{1}{\pi a^2}
  \sum_{m=-\infty}^{\infty} \cos m(\t-\t') \sum_{n=1}^{\infty} e^{-D_3\amn^2t} 
  \frac{J_m(\amn r)J_m(\amn r')}
  {[J'_m(\amn a)]^2},
\end{align}

where $\amn$ is the $n$-th root of
\begin{align}
  J_m(a\a) = 0.
\end{align}
