\paragraph{Green's function}
\begin{align}
  v(r,\t, t) = \frac{1}{2\pi}
  \sum_{m=-\infty}^{\infty} \cos m(\t-\t')
  \int_0^{\infty} C_m(ur)C_m(ur')u e^{-D_3 u^2 t} du,
\end{align}

where
\begin{align}
  C_m(z) =
  \frac{ J_m(z) [uY_m'(au) - hY_m(au)] -
	 Y_m(z) [uJ_m'(au) - hJ_m(au)] }
       { \{ [u J_m'(au) - hJ_m(au)]^2 +
	    [u Y_m'(au) - hY_m(au)]^2  \}^\frac{1}{2} }.
\end{align}
